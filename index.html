function solve() { const m = M_rows, n = M_cols; const out = document.getElementById('output'); out.innerHTML = ''; out.style.display = 'block'; // 🔥 قراءة المصفوفة مرة واحدة (سريع) let M = new Array(m); for (let i = 0; i < m; i++) { M[i] = new Float64Array(n + 1); for (let j = 0; j < n; j++) { M[i][j] = +document.getElementById(`m_${i}_${j}`).value || 0; } M[i][n] = +document.getElementById(`b_${i}`).value || 0; } // عرض المعادلات out.innerHTML += showEquations(); // 🔥 Gaussian سريع let row = 0; for (let col = 0; col < n && row < m; col++) { // pivot let sel = row; for (let i = row + 1; i < m; i++) { if (Math.abs(M[i][col]) > Math.abs(M[sel][col])) sel = i; } if (Math.abs(M[sel][col]) < EPS) continue; // swap if (sel !== row) { let tmp = M[sel]; M[sel] = M[row]; M[row] = tmp; } // eliminate for (let i = row + 1; i < m; i++) { let factor = M[i][col] / M[row][col]; if (Math.abs(factor) < EPS) continue; for (let j = col; j <= n; j++) { M[i][j] -= factor * M[row][j]; } } row++; } // 🔥 rank سريع let rankA = 0, rankAug = 0; for (let i = 0; i < m; i++) { let allZero = true; for (let j = 0; j < n; j++) { if (Math.abs(M[i][j]) > EPS) { allZero = false; break; } } if (!allZero) { rankA++; rankAug++; } else if (Math.abs(M[i][n]) > EPS) { rankAug++; } } // 🔥 تحديد النوع بسرعة let type; if (rankA !== rankAug) type = "No Solution"; else if (rankA === n) type = "Unique Solution"; else type = "Infinite Solutions"; // 🔥 output مرة واحدة (أسرع) let html = `
Result
${rankA}
r(A)
${rankAug}
r(A|b)
${n}
n
${type}
`; // unique solution if (type === "Unique Solution") { let sol = new Float64Array(n); for (let i = rankA - 1; i >= 0; i--) { let pivotCol = -1; for (let j = 0; j < n; j++) { if (Math.abs(M[i][j]) > EPS) { pivotCol = j; break; } } let sum = M[i][n]; for (let j = pivotCol + 1; j < n; j++) { sum -= M[i][j] * sol[j]; } sol[pivotCol] = sum / M[i][pivotCol]; } html += `
`; for (let i = 0; i < n; i++) { html += `
x${i+1} = ${fmt(sol[i])}
`; } html += `
`; } // infinite if (type === "Infinite Solutions") { html += `
Free variables = ${n - rankA}
`; } html += `
`; out.innerHTML += html; }
