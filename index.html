function solve() { const m = M_rows, n = M_cols; const out = document.getElementById('output'); // قراءة سريعة let M = []; for (let i = 0; i < m; i++) { M[i] = []; for (let j = 0; j < n; j++) { M[i][j] = +document.getElementById(`m_${i}_${j}`).value || 0; } M[i][n] = +document.getElementById(`b_${i}`).value || 0; } // Gaussian (خفيف جدًا) let r = 0; for (let c = 0; c < n && r < m; c++) { let piv = r; for (let i = r; i < m; i++) { if (Math.abs(M[i][c]) > Math.abs(M[piv][c])) piv = i; } if (Math.abs(M[piv][c]) < 1e-9) continue; [M[r], M[piv]] = [M[piv], M[r]]; for (let i = r + 1; i < m; i++) { let f = M[i][c] / M[r][c]; for (let j = c; j <= n; j++) { M[i][j] -= f * M[r][j]; } } r++; } // rank let rankA = 0, rankAug = 0; for (let i = 0; i < m; i++) { let zero = true; for (let j = 0; j < n; j++) { if (Math.abs(M[i][j]) > 1e-9) { zero = false; break; } } if (!zero) { rankA++; rankAug++; } else if (Math.abs(M[i][n]) > 1e-9) { rankAug++; } } // نوع الحل let type; if (rankA !== rankAug) type = "No Solution"; else if (rankA === n) type = "Unique Solution"; else type = "Infinite Solutions"; // حل وحيد let solText = ""; if (type === "Unique Solution") { let x = new Array(n).fill(0); for (let i = rankA - 1; i >= 0; i--) { let pc = -1; for (let j = 0; j < n; j++) { if (Math.abs(M[i][j]) > 1e-9) { pc = j; break; } } let sum = M[i][n]; for (let j = pc + 1; j < n; j++) { sum -= M[i][j] * x[j]; } x[pc] = sum / M[i][pc]; } for (let i = 0; i < n; i++) { solText += `
x${i+1} = ${x[i].toFixed(3)}
`; } } // عرض خفيف جدًا out.innerHTML = `
Result
${rankA}
r(A)
${rankAug}
r(A|b)
${n}
n
${type}
${solText} ${type === "Infinite Solutions" ? `
Free variables = ${n - rankA}
` : ""}
`; out.style.display = "block"; }
